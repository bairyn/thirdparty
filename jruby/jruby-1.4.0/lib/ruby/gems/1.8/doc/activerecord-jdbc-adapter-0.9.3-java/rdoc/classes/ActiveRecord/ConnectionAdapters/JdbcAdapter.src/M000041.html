      def pk_and_sequence_for(table)
        result_set = @connection.connection.get_meta_data.get_primary_keys(nil, nil, table)
        if result_set.next
          keys = [result_set.getString("COLUMN_NAME"), nil]
        end
        keys.blank? ? nil : keys
      ensure
        result_set.close
      end
