      def load_ruby_prof
        begin
          gem 'ruby-prof', '>= 0.6.1'
          require 'ruby-prof'
          if mode = options[:measure]
            RubyProf.measure_mode = RubyProf.const_get(mode.upcase)
          end
        rescue LoadError
          abort '`gem install ruby-prof` to use the profiler'
        end
      end
