          def expand_sources
            if @sources.include?(:all)
              assets = all_asset_files
              ((defaults.dup & assets) + assets).uniq!
            else
              expanded_sources = validate_sources!
              expanded_sources << APPLICATION_JS if include_application?
              expanded_sources
            end
          end
