          def write_asset_file_contents(joined_asset_path)
            FileUtils.mkdir_p(File.dirname(joined_asset_path))
            File.open(joined_asset_path, "w+") { |cache| cache.write(joined_contents) }
            mt = latest_mtime
            File.utime(mt, mt, joined_asset_path)
          end
